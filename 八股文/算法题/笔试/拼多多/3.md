标准坐标系，给定一个目标坐标x,y，还有一个操作序列（上下左右），问删除操作序列删除一个最短连续子序列能让这个序列到目标，这个最短长度多少

思路：建立**前缀和**，快速获取一个子序列的位移，总位移减去子序列位移等于目标即可。同时，因为总位移和目标已知，那么就**只遍历子序列右端点**，用哈希表存储起点到左端点的结果，获取符合条件最右的左端点即可

```java
import java.io.*;
import java.util.*;

public class pdd_3{
    public static void main(String[] args) throws Exception {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StreamTokenizer st = new StreamTokenizer(br);
        st.nextToken(); int T = (int)st.nval;
        StringBuilder print = new StringBuilder();
        while(T--!=0){
            st.nextToken(); int n = (int)st.nval;
        	st.nextToken(); String s = st.sval;
            st.nextToken(); int x = (int)st.nval;
            st.nextToken(); int y = (int)st.nval;
            int[][] preSum = new int[n+1][2];
            // 前缀和对索引
            Map<String, Integer> map = new HashMap<>();
            int[] init = new int[]{0,0};
            map.put(Arrays.toString(init),0);
            int ans = n;
            for(int i = 0;i<n;i++){
                char op = s.charAt(i);
                preSum[i+1][0] = preSum[i][0];
                preSum[i+1][1] = preSum[i][1];
                if(op=='U'){
                    preSum[i+1][0]++;
                } else if(op=='D'){
                    preSum[i+1][0]--;
                } else if(op=='L'){
                    preSum[i+1][1]--;
                } else if(op=='R'){
                    preSum[i+1][1]++;
                }
                // [l,r] = preSum[r]-preSum[l-1] = x,y
            }
            int sumX = preSum[n][0], sumY = preSum[n][1];
            int targetX = sumX-x, targetY = sumY-y;
            System.out.println("区间目标"+targetX+","+targetY);
            if(targetX==0 && targetY==0){
                print.append(0+"\n");
                continue;
            }
            for(int i = 1;i<=n;i++){
                System.out.println("现在的前缀和"+preSum[i][0]+","+preSum[i][1]);
                int findX = preSum[i][0]-targetX;
                int findY = preSum[i][1]-targetY;
                int[] tmp = new int[]{findX, findY};
                Integer index = map.get(Arrays.toString(tmp));
                map.put(Arrays.toString(preSum[i]),i);
                if(index==null) continue;
                ans = Math.min(ans, i-index);
            }
            ans=ans==n?-1:ans;
            print.append(ans+"\n");
        }
        System.out.println(print);
    }
}
```
