有n个加油站，分别距离起点d，充电单价p，小明电动车续航C，要到达距离L的终点。问最少花多少钱能到达终点

思路：获取每个站点下一个更便宜的站点索引。遍历站点，每次开车到一个站点，扣减能量。如果说下个便宜站点充满电也无法到达就直接在这里充满电，如果充电能到就充到刚好能到，如果能直接到就不充电

```java
import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.io.StreamTokenizer;
import java.util.*;

public class pdd_2 {
    public static void main(String[] args) throws Exception{
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StreamTokenizer st = new StreamTokenizer(br);
        st.nextToken(); int L = (int)st.nval;
        st.nextToken(); int C = (int)st.nval;
        st.nextToken(); int n = (int)st.nval;
        int[] d = new int[n+1];
        int[] p = new int[n+1];
        for(int i = 0;i<n;i++){
            st.nextToken();d[i] = (int)st.nval;
            st.nextToken();p[i] = (int)st.nval;
        }
        d[n] = L; p[n] = 0;
        // 获取每个充电站下一个更便宜的索引
        int[] next = new int[n+1];
        Arrays.fill(next,n);
        Stack<Integer> stk = new Stack<>();
        for(int i = 0;i<=n;i++){
            while(!stk.isEmpty() && p[i]<p[stk.peek()]){
                int pre = stk.pop();
                next[pre] = i;
            }
            stk.push(i);
        }
        int energy = C, price = 0;
        if(L<=energy){
            System.out.println(0);
            return ;
        }
        for(int i = 0;i<n;i++){
            // System.out.println("到达站点"+i);
            energy -= d[i]-((i==0)?0:d[i-1]);
            if(energy<0){
                System.out.println("-1");
                return;
            }
            int cheap = next[i];
            int distance = d[cheap]-d[i];
            if(distance>C){
                // 充满
                price += (C-energy)*p[i];
                energy = C;
            } else if(energy>=distance){
                continue;
            } else {
                price += (distance-energy)*p[i];
                energy = distance;
            }
        }
        System.out.println(price);
    }
}
```
