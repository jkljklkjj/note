每天要求算力need。可以选择开算力包，生效w天，最多开m个。计算一个算力包最少多少算力能满足每天的需要

思路：首先，**一旦出现算力不足的天，就把算力包开到刚好满足**，是回报最高的操作，这是贪心策略。然后可以通过差分数组**快速进行区间加减**，然后用**一个指针能随时获取到差分数组还原后的结果**，这个神奇操作。然后**二分算力包算力**，因为算力越高越能满足需求的单调性

```java
import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.io.StreamTokenizer;

public class pdd_4 {
    public static void main(String[] args) throws Exception{
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StreamTokenizer st = new StreamTokenizer(br);
        StringBuilder print = new StringBuilder();
        st.nextToken(); int T = (int)st.nval;
        while(T--!=0){
            st.nextToken(); int n = (int)st.nval;
            st.nextToken(); int m = (int)st.nval;// 最大数目
            st.nextToken(); int w = (int)st.nval;// 生效时长
            int[] need = new int[n];
            int max = 0;
            for(int i = 0;i<n;i++){
                st.nextToken(); need[i] = (int)st.nval;
            }
            for(int i = 0;i<n;i++){
                st.nextToken(); need[i] -= (int)st.nval;
                max = Math.max(max,need[i]);
            }
            if(max<=0){
                print.append("0 0");
                continue;
            }
            int l = 1, r = max;
            while(l<=r){
                int mid = l+(r-l)/2;
                int[] res = check(mid,w,need,m,n);
                if(res[0]==1){
                    r = mid-1;
                } else {
                    l = mid+1;
                }
            }
            int[] res = check(l,w,need,m,n);
            if(res[1]==0) {
                print.append("-1\n");
            }
            else print.append(l+" "+res[1]+"\n");
        }
        System.out.println(print);
    }
    public static int[] check(int x, int w, int[] need, int m, int n){
        int[] add = new int[n+w];// 开通的包数的差分
        int curM = 0;// 目前开通的包数
        long now = 0;
        for(int i = 0;i<n;i++){
            now += add[i]*x;
            if(now<need[i]){
                // 开通对应的包
                int needBag = (int)(need[i]-now+x)/x;
                add[i+w]-=needBag;
                now+=needBag*x;
                curM += needBag;
                if(curM>m) return new int[]{0,0};
            }
        }
        return new int[]{1,curM};
    }
}
```
