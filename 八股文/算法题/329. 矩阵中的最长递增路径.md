给定一个矩阵，路径支持左上右下走，求出经过格子顺序递增的最长路径的长度

思路：进行一个dfs。但是如果dfs到同一个点那么答案肯定是一样的，不需要再搜索。所以引入记忆化。同时对于这类题目一定要把dfs的返回值设置成int，可以在顶层节点获取到结果而不是传到叶子节点的时候再确认答案，这样逻辑通顺

```java
int[][] dirs = {{-1,0},{1,0},{0,-1},{0,1}};
int n, m;

public int longestIncreasingPath(int[][] matrix) {
	n = matrix.length;
	m = matrix[0].length;
	int[][] memo = new int[n][m];
	int ans = 0;
	for(int i = 0;i<n;i++){
		for(int j = 0;j<m;j++){
			ans = Math.max(ans, dfs(matrix,i,j,memo));
		}
	}
	return ans;
}

public int dfs(int[][] matrix, int i, int j, int[][] memo){
	if(memo[i][j]!=0) return memo[i][j];
	memo[i][j]++;
	for(int[] dir:dirs){
		int x = i+dir[0], y = j+dir[1];
		if(check(x,y) && matrix[x][y]>matrix[i][j]){
			memo[i][j] = Math.max(memo[i][j], dfs(matrix,x,y,memo)+1);
		}
	}
	return memo[i][j];
}

public boolean check(int i, int j){
	if(i>=0 && j>=0 && i<n && j<m) return true;
	return false;
}
```
