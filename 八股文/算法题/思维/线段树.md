给定一个数组，求任意区间的结果，同时支持多次修改。这个时候要用上线段树，一个节点代表一个区间的结果，这些结果可以是真正的结果，也可以是中间结果。然后他的孩子节点表示它分成两半的结果。求区间结果的时候通过子区间结果计算即可

```java
class TreeNode {
	// 这个节点维护的信息
	int val1, val2;
	// ...
	public TreeNode(int val1, int val2...){
		this.val1 = val1;
		this.val2 = val2;
		// ...
	}
}

class SegmentTree {
    TreeNode[] tree;
    int n;

    public SegmentTree(int[] nums) {
        n = nums.length;
        tree = new int[4 * n];
        // 1为根节点
        build(nums, 1, 0, n - 1);
    }

    void build(int[] nums, int node, int l, int r) {
        if (l == r) {
	        // 只有一个数的情况
            tree[node] = nums[l];
            return;
        }
        int mid = (l + r) >>> 1;
        // 对半切数组，递归左节点2i，右节点2i+1
        // 获取子节点的结果
        int l = build(nums, node << 1, l, mid);
        int r build(nums, node << 1 | 1, mid + 1, r);
        // 通过子节点结果求根的结果
        merge(node, l, r);
    }

	// 更新这个区间的结果
    void update(int pos, int val, int node, int l, int r) {
        if (l == r) {
	        // 一个数的更新逻辑
            tree[node] = val;
            return;
        }
        int mid = (l + r) >>> 1;
        // 要更新的数在左节点
        if (pos <= mid) update(pos, val, node << 1, l, mid);
        else update(pos, val, node << 1 | 1, mid + 1, r);
        tree[node] = tree[node << 1] + tree[node << 1 | 1];
    }

	// 查找区间的结果
    TreeNode query(int ql, int qr, int node, int l, int r) {
        if (ql <= l && r <= qr) return tree[node];
        int mid = (l + r) >>> 1;
        TreeNode res = new TreeNode();
        // 合并子数组的结果
        if (ql <= mid) res += query(ql, qr, node << 1, l, mid);
        if (qr > mid) res += query(ql, qr, node << 1 | 1, mid + 1, r);
        return res;
    }
}
```

```java
class SegmentTree {
    int[] tree; // 存储这个节点代表的区间和
    int n;

    public SegmentTree(int[] nums) {
        n = nums.length;
        tree = new int[4 * n];
        // 1为根节点
        build(nums, 1, 0, n - 1);
    }

    void build(int[] nums, int node, int l, int r) {
        if (l == r) {
	        // 一个数的区间和等于它自己
            tree[node] = nums[l];
            return;
        }
        int mid = (l + r) >>> 1;
        // 对半切数组，递归左节点2i，右节点2i+1
        build(nums, node << 1, l, mid);
        build(nums, node << 1 | 1, mid + 1, r);
        // 根节点的区间和等于两部分区间的区间和
        tree[node] = tree[node << 1] + tree[node << 1 | 1];
    }

    void update(int pos, int val, int node, int l, int r) {
        if (l == r) {
	        // 更新该节点的值
            tree[node] = val;
            return;
        }
        int mid = (l + r) >>> 1;
        // 如果要更新的点在左子树
        if (pos <= mid) update(pos, val, node << 1, l, mid);
        else update(pos, val, node << 1 | 1, mid + 1, r);
        // 更新后也更新这个节点的结果
        tree[node] = tree[node << 1] + tree[node << 1 | 1];
    }

    int query(int ql, int qr, int node, int l, int r) {
	    // 如果区间包含于查询区间
        if (ql <= l && r <= qr) return tree[node];
        int mid = (l + r) >>> 1;
        int res = 0;
        // 合并两个孩子的结果
        // 如果孩子和区间有交集
        if (ql <= mid) res += query(ql, qr, node << 1, l, mid);
        if (qr > mid) res += query(ql, qr, node << 1 | 1, mid + 1, r);
        return res;
    }
}
```

同时线段树支持懒标记，修改区间不直接修改叶子节点，而是每次标记一个完全被包含于修改区间的区间，被标记的区间无交集而且并集等于修改区间
每个节点不仅有它的区间和，还有它的懒标记。修改时如果说遍历到的节点区间包含于修改区间，就直接修改区间和和懒标记，如果没有就下推懒标记。下推时更新左右孩子的区间和并传递懒标记

```java
class SegmentTree {
    int[] tree; // 存储区间和
    int[] lazy; // 懒标记（待加的值）
    int n;

    public SegmentTree(int[] nums) {
        n = nums.length;
        tree = new int[4 * n];
        lazy = new int[4 * n];
        build(nums, 1, 0, n - 1);
    }

    void build(int[] nums, int node, int l, int r) {
        if (l == r) {
            tree[node] = nums[l];
            return;
        }
        int mid = (l + r) >>> 1;
        build(nums, node << 1, l, mid);
        build(nums, node << 1 | 1, mid + 1, r);
        tree[node] = tree[node << 1] + tree[node << 1 | 1];
    }

    // 下推懒标记
    void pushDown(int node, int l, int r) {
        if (lazy[node] != 0) {
            int mid = (l + r) >>> 1;
            int left = node << 1;
            int right = node << 1 | 1;

            // 更新左右子节点的值
            tree[left] += lazy[node] * (mid - l + 1);
            tree[right] += lazy[node] * (r - mid);

            // 传递懒标记
            lazy[left] += lazy[node];
            lazy[right] += lazy[node];

            // 清除当前节点标记
            lazy[node] = 0;
        }
    }

    // 单点修改
    void update(int pos, int val, int node, int l, int r) {
        if (l == r) {
            tree[node] = val;
            return;
        }
        pushDown(node, l, r); // 单点修改也可能需要下推（如果之前有懒标记）
        int mid = (l + r) >>> 1;
        if (pos <= mid) update(pos, val, node << 1, l, mid);
        else update(pos, val, node << 1 | 1, mid + 1, r);
        tree[node] = tree[node << 1] + tree[node << 1 | 1];
    }

    // 区间修改 [ul, ur] 每个数 +val
    void updateRange(int ul, int ur, int val, int node, int l, int r) {
        if (ul <= l && r <= ur) {
            // 完全覆盖，直接打标记
            tree[node] += val * (r - l + 1);
            lazy[node] += val;
            return;
        }

        pushDown(node, l, r); // 部分覆盖，必须下推
        
        int mid = (l + r) >>> 1;
        if (ul <= mid) updateRange(ul, ur, val, node << 1, l, mid);
        if (ur > mid) updateRange(ul, ur, val, node << 1 | 1, mid + 1, r);

        tree[node] = tree[node << 1] + tree[node << 1 | 1];
    }

    // 区间查询（你原来的）
    int query(int ql, int qr, int node, int l, int r) {
        if (ql <= l && r <= qr) return tree[node];

        pushDown(node, l, r); // 查询也可能需要下推

        int mid = (l + r) >>> 1;
        int res = 0;
        if (ql <= mid) res += query(ql, qr, node << 1, l, mid);
        if (qr > mid) res += query(ql, qr, node << 1 | 1, mid + 1, r);
        return res;
    }
}
```
